88int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t *bcd_number) {
89 int max_value = 1;
90 for (int i = 0; i < digits; i++) {
91 max_value *= 10;
92 }
93
94 if (data >= max_value) {
95 printf("Error: El número %lu no cabe en %d dígitos\n", (unsigned long)data, digits);
96 return -1; // Error
97 }
98
99 // Convertimos el número a BCD y almacenamos cada dígito en el arreglo
100 for (int i = 0; i < digits; i++) {
101 bcd_number[digits - 1 - i] = data % 10;
102 printf("Guardando %d en bcd_number[%d]\n", bcd_number[digits - 1 - i], digits - 1 - i);
103 data /= 10;
104 }
105
106 printf("Arreglo final BCD: ");
107 for (int i = 0; i < digits; i++) {
108 printf("%d ", bcd_number[i]);
109 }
110 printf("\n");
111
112 return 0; // Éxito
113}

134void DisplayNumberOnLcd(uint32_t data, uint8_t digits, gpioConf_t *bcd_gpio, gpioConf_t *sel_gpio)
135{
136 uint8_t bcd_array[digits];
137 if (convertToBcdArray(data, digits, bcd_array) != 0) {
138 printf("DisplayNumberOnLcd: Error en la conversión a BCD\n");
139 return;
140 }
141
142 for (int i = 0; i < digits; i++) {
143 printf("Mostrando dígito %d (valor %d) en el display, seleccionando pin %d\n", i, bcd_array[i], sel_gpio[i].pin);
144 // Selecciona el dígito a mostrar (activar el pin correspondiente)
145 for (int j = 0; j < digits; j++) {
146 if (j == i) {
147 GPIOOn(sel_gpio[j].pin);
148 printf(" Activando selección de dígito en pin %d\n", sel_gpio[j].pin);
149 } else {
150 GPIOOff(sel_gpio[j].pin);
151 }
152 }
153
154 // Envía el valor BCD al display usando la función reutilizada
155 BCDtoGPIO(bcd_array[i], bcd_gpio);
156 }
157}
